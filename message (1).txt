        if LoopElement["Character"] then

        end
    end

end
local function FunctionArgument()
    table.clear(EmptyTable)
    local FunctionResult = FunctionParameter()
    if not FunctionResult then
        return nil
    end
    for LoopItem, LoopObject in pairs(FunctionResult) do
        if (LoopObject and LoopObject:IsA("Model")) then
            local HumanoidComponent = LoopObject:FindFirstChildOfClass("Humanoid")
            local RootPartComponent = LoopObject:FindFirstChild("HumanoidRootPart")
            if
                (HumanoidComponent and (HumanoidComponent["Health"] > 0) and RootPartComponent and v12["Character"] and
                    v12["Character"]:FindFirstChild("HumanoidRootPart"))
             then
                if ((RootPartComponent["Position"] - v12["Character"]["HumanoidRootPart"]["Position"])["Magnitude"] < 100) then
                    table.insert(EmptyTable, RootPartComponent)
                end
            end
        end
    end
    if (#EmptyTable == 0) then
        return nil
    end
    return EmptyTable
end
local function FunctionValue()
    local FunctionOutput = FunctionArgument()
    if not FunctionOutput then
        return
    end
    local TableElement = FunctionOutput[1]
    if not TableElement then
        return
    end
    table.clear(EmptyTable)
    for LoopCounterValue = 2, #FunctionOutput do
        local TableItem = FunctionOutput[LoopCounterValue]
        local ParentObject = TableItem["Parent"]
        if ParentObject then
            table.insert(EmptyTable, {ParentObject, TableItem})
        end
    end
    RegisterAttackFunction:FireServer(NaN)
    coroutine.resume(NetModule, TableElement, EmptyTable)
end
