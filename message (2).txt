-- Noclip Loop Function
local function NoclipLoop()
    if LocalPlayer.Character ~= nil then
        for _, child in pairs(LocalPlayer.Character:GetDescendants()) do
            if child:IsA("BasePart") and child.CanCollide == true then
                child.CanCollide = false
            end
        end
    end
end

-- Function to manage noclip connection
local function UpdateNoclip()
    if IsFarmingActive() then
        -- Start noclip if not already running
        if not Noclipping then
            Noclipping = RunService.Stepped:Connect(NoclipLoop)
        end
    else
        -- Stop noclip if running
        if Noclipping then
            Noclipping:Disconnect()
            Noclipping = nil
        end
    end
end

-- Initial check

-- Improved TP function for islands (no shaking)
function TPToIsland(targetCFrame)
    if not LocalPlayer.Character or not LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
        return
    end
    
    local hrp = LocalPlayer.Character.HumanoidRootPart
    local distance = (targetCFrame.Position - hrp.Position).Magnitude
    
    -- If already very close, just set position and stop
    if distance <= 100 then
        -- Clean up ALL body movers first
        for _, obj in pairs(hrp:GetChildren()) do
            if obj:IsA("BodyVelocity") or obj:IsA("BodyGyro") then
                obj:Destroy()
            end
        end
        
        hrp.CFrame = targetCFrame
        hrp.Velocity = Vector3.new(0, 0, 0)
        hrp.RotVelocity = Vector3.new(0, 0, 0)
        
        return true  -- Arrived
    end
    
    -- Clean up old tween if exists
    if hrp:FindFirstChild("BodyClip") then
        hrp:FindFirstChild("BodyClip"):Destroy()
    end
    
    -- Create smooth tween
    local tweenInfo = TweenInfo.new(
        distance / 300, 
        Enum.EasingStyle.Linear,
        Enum.EasingDirection.Out
    )
    
    local tween = TweenService:Create(hrp, tweenInfo, {CFrame = targetCFrame})
    tween:Play()
    
    return false  -- Still moving
end





function StopTween()
    if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
        local hrp = LocalPlayer.Character.HumanoidRootPart
        
        -- Remove ALL body movers
        for _, obj in pairs(hrp:GetChildren()) do
            if obj:IsA("BodyVelocity") or obj:IsA("BodyGyro") or obj.Name == "BodyClip" or obj.Name == "AntiShake" then
                obj:Destroy()
            end
        end
        
        -- Stop all velocity
        hrp.Velocity = Vector3.new(0, 0, 0)
        hrp.RotVelocity = Vector3.new(0, 0, 0)
        
        wait(0.1)
    end
end

local function cleanupNoClip()
    pcall(function()
        -- Clean up the RunService connection (Infinite Yield style)
        if noclipConnection then
            noclipConnection:Disconnect()
            noclipConnection = nil
        end
        
        -- Clean up body movers
        if bodyClipInstance then
            bodyClipInstance:Destroy()
            bodyClipInstance = nil
        end
        
        -- Re-enable collisions for all parts
        if LocalPlayer.Character then
            for _, v in pairs(LocalPlayer.Character:GetDescendants()) do
                if v:IsA("BasePart") then
                    v.CanCollide = true
                end
            end
        end
        
        -- Also clean up any other physics objects
        if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
            local hrp = LocalPlayer.Character.HumanoidRootPart
            for _, obj in pairs(hrp:GetChildren()) do
                if obj:IsA("BodyVelocity") or obj:IsA("BodyGyro") or obj.Name == "BodyClip" then
                    obj:Destroy()
                end
            end
        end
    end)
end

local function setupNoClip()
    pcall(function()
        if not LocalPlayer.Character or not LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
            return
        end
        
        local hrp = LocalPlayer.Character.HumanoidRootPart
        
        -- Clean up any existing noclip first
        cleanupNoClip()
        
        -- 1. Set up Infinite Yield style noclip (main part)
        noclipConnection = game:GetService("RunService").Stepped:Connect(noclipLoop)
        
        -- 2. Create BodyClip for movement control (keep your existing setup)
        bodyClipInstance = Instance.new("BodyVelocity")
        bodyClipInstance.Name = "BodyClip"
        bodyClipInstance.Parent = hrp
        bodyClipInstance.MaxForce = Vector3.new(10000, 10000, 10000)
        bodyClipInstance.Velocity = Vector3.new(0, 0, 0)
        
        -- 3. Immediately set CanCollide to false on all parts (Infinite Yield style)
        for _, v in pairs(LocalPlayer.Character:GetDescendants()) do
            if v:IsA("BasePart") then
                v.CanCollide = false
            end
        end
    end)
end

-- Function to check if Island exists

    if game:GetService("Workspace")["_WorldOrigin"].Locations:FindFirstChild("Island " .. islandNumber) then

-- TP function (with anti-spam)
local lastTPTime = 0
function TP(targetCFrame)
    if not LocalPlayer.Character or not LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
        return
    end
    
    local hrp = LocalPlayer.Character.HumanoidRootPart
    local distance = (targetCFrame.Position - hrp.Position).Magnitude
    
    -- Skip if very close OR if we just teleported recently
    if distance <= State.SkipDistance then
        hrp.CFrame = targetCFrame
        hrp.Velocity = Vector3.new(0, 0, 0)
        return
    end
    
    -- Prevent spam teleports (minimum 0.1 second between calls)
    if tick() - lastTPTime < 0.1 then
        return
    end
    lastTPTime = tick()
    
    -- Clean old BodyClip
    if hrp:FindFirstChild("BodyClip") then
        hrp:FindFirstChild("BodyClip"):Destroy()
    end
    
    -- Tween
    local tweenInfo = TweenInfo.new(distance / 300, Enum.EasingStyle.Linear)
    local tween = TweenService:Create(hrp, tweenInfo, {
        CFrame = targetCFrame
    })
    
    tween:Play()
end

-- Function to get closest chest
local function getClosestChest()
    local closestChest = nil
    local closestDistance = math.huge
    
    if not LocalPlayer.Character or not LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
        return nil
    end
    
    local playerPos = LocalPlayer.Character.HumanoidRootPart.Position
    
    if Workspace:FindFirstChild("ChestModels") then
        for _, chestType in pairs(Workspace.ChestModels:GetChildren()) do
            for _, chest in pairs(chestType:GetChildren()) do
                if chest:IsA("Model") or chest:IsA("Part") then
                    local chestPos
                    if chest:IsA("Model") and chest.PrimaryPart then
                        chestPos = chest.PrimaryPart.Position
                    elseif chest:IsA("Part") then
                        chestPos = chest.Position
                    end
                    
                    if chestPos then
                        local distance = (chestPos - playerPos).Magnitude
                        if distance < closestDistance then
                            closestDistance = distance
                            closestChest = chest
                        end
                    end
                end
            end
        end
    end
    
    return closestChest
end

-- Auto Chest Loop
spawn(function()
    while wait(0.1) do
        if State.AutoChest then
            pcall(function()
                local chest = getClosestChest()
                if chest then
                    local targetCFrame
                    if chest:IsA("Model") and chest.PrimaryPart then
                        targetCFrame = chest.PrimaryPart.CFrame
                    elseif chest:IsA("Part") then
                        targetCFrame = chest.CFrame
                    end
                    
                    if targetCFrame and LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
                        local hrp = LocalPlayer.Character.HumanoidRootPart
                        local distance = (targetCFrame.Position - hrp.Position).Magnitude
                        
                        if distance > 5 then
                            local tweenInfo = TweenInfo.new(distance / 500, Enum.EasingStyle.Linear)
                            local tween = TweenService:Create(hrp, tweenInfo, {CFrame = targetCFrame})
                            tween:Play()
                        end
                    end
                end
            end)
        end
    end
end)

-- Helper: checks if "Fruit" is the 2nd or 3rd word in the name
local function isFruit(name)
    local words = {}
    for word in string.gmatch(name, "%S+") do
        table.insert(words, word)
    end
    return words[2] == "Fruit" or words[3] == "Fruit"
end

-- Helper: returns all fruit tools from Backpack AND workspace.Characters.(USERNAME)
local function getAllFruits()
    local fruits = {}

    -- Check Backpack
    for _, v in pairs(LocalPlayer.Backpack:GetChildren()) do
        if isFruit(v.Name) then
            table.insert(fruits, v)
        end
    end

    -- Check workspace.Characters.(USERNAME)
    local charactersFolder = workspace:FindFirstChild("Characters")
    if charactersFolder then
        local myChar = charactersFolder:FindFirstChild(LocalPlayer.Name)
        if myChar then
            for _, v in pairs(myChar:GetChildren()) do
                if isFruit(v.Name) then
                    table.insert(fruits, v)
                end
            end
        end
    end

    return fruits
end

-- ===================== AUTO BUSO HAKI SYSTEM =====================
-- Proper Auto Buso that checks HasBuso value before activating

local AutoBusoEnabled = true
local BusoCooldown = true

-- Function to check if Buso Haki is currently active (PROPER CHECK)
function IsBusoActive()
    if LocalPlayer.Character then
        -- Check if HasBuso value exists in character
        local hasBusoValue = LocalPlayer.Character:FindFirstChild("HasBuso")
        if hasBusoValue then
            return true
        end
    end
    return false
end

-- Main Auto Buso Loop
spawn(function()
    while task.wait(0.5) do  -- Check every 0.5 seconds
        if AutoBusoEnabled and not BusoCooldown then
            pcall(function()
                -- Only activate Buso if HasBuso doesn't exist
                if not IsBusoActive() then
                    BusoCooldown = true
                    
                    -- Debug message (optional)
                    -- print("[Auto Buso] Activating Buso Haki...")
                    
                    -- Activate Buso Haki
                    ReplicatedStorage.Remotes.CommF_:InvokeServer("Buso")
                    
                    -- Wait for activation to register
                    task.wait(0.2)
                    BusoCooldown = false
                end
            end)
        end
    end
end)

-- Character respawn handler to re-enable Buso
LocalPlayer.CharacterAdded:Connect(function(character)
    task.wait(0.5)  -- Wait for character to load
    if AutoBusoEnabled then
        task.wait(1)  -- Additional wait for character to fully initialize
        -- Check if HasBuso exists after respawn
        task.wait(0.3)
        if not character:FindFirstChild("HasBuso") then
            ReplicatedStorage.Remotes.CommF_:InvokeServer("Buso")
        end
    end
end)

-- Also monitor when Buso wears off
spawn(function()
    while task.wait() do
        if AutoBusoEnabled and LocalPlayer.Character then
            -- Continuously monitor for HasBuso
            local hasBuso = LocalPlayer.Character:FindFirstChild("HasBuso")
            if not hasBuso then
                -- Buso has worn off, re-activate it
                ReplicatedStorage.Remotes.CommF_:InvokeServer("Buso")
            end
        end
    end
end)

        local dragInput
        local dragStart
        local startPos
        local hasMoved = false
        
        local function update(input)
            local delta = input.Position - dragStart
            container.Position = UDim2.new(
                startPos.X.Scale,
                startPos.X.Offset + delta.X,
                startPos.Y.Scale,
                startPos.Y.Offset + delta.Y
            )
            
            -- Track if we've moved significantly
            if math.abs(delta.X) > 5 or math.abs(delta.Y) > 5 then
                hasMoved = true
            end
        end
        
        -- Start dragging when mouse is pressed on button
        button.InputBegan:Connect(function(input)
            if input.UserInputType == Enum.UserInputType.MouseButton1 then
                dragging = true
                dragStart = input.Position
                startPos = container.Position
                hasMoved = false
                
                input.Changed:Connect(function()
                    if input.UserInputState == Enum.UserInputState.End then
                        dragging = false
                    end
                end)
            end
        end)
        
        -- Update position while dragging
        button.InputChanged:Connect(function(input)
            if input.UserInputType == Enum.UserInputType.MouseMovement then
                dragInput = input
            end
        end)
        
        UIS.InputChanged:Connect(function(input)
            if dragging and input == dragInput then
                update(input)
            end
        end)
        
        -- Click to toggle UI (only if not dragging)
        button.MouseButton1Click:Connect(function()
            -- Only trigger if we didn't drag
            if not hasMoved then
                -- Press down animation
                local shrinkTween = TweenService:Create(button, TweenInfo.new(
                    0.1, 
                    Enum.EasingStyle.Quad, 
                    Enum.EasingDirection.Out
                ), {
                    Size = UDim2.fromOffset(50, 50)  -- Adjusted for 80x80 container
                })
                
                local bgShrink = TweenService:Create(background, TweenInfo.new(
                    0.1, 
                    Enum.EasingStyle.Quad, 
                    Enum.EasingDirection.Out
                ), {
                    Size = UDim2.fromOffset(62, 62)  -- Adjusted for 80x80 container
                })
                
                local strokeShrink = TweenService:Create(strokeContainer, TweenInfo.new(
                    0.1, 
                    Enum.EasingStyle.Quad, 
                    Enum.EasingDirection.Out
                ), {
                    Size = UDim2.fromOffset(57, 57)  -- Adjusted for 80x80 container
                })
                
                local innerShrink = TweenService:Create(innerCircle, TweenInfo.new(
                    0.1, 
                    Enum.EasingStyle.Quad, 
                    Enum.EasingDirection.Out
                ), {
                    Size = UDim2.fromOffset(50, 50)  -- Adjusted for 80x80 container
                })
                
                shrinkTween:Play()
                bgShrink:Play()
                strokeShrink:Play()
                innerShrink:Play()
                shrinkTween.Completed:Wait()
                
                -- Bounce back animation
                local bounceTween = TweenService:Create(button, TweenInfo.new(
                    0.4, 
                    Enum.EasingStyle.Elastic, 
                    Enum.EasingDirection.Out
                ), {
                    Size = UDim2.fromOffset(60, 60)  -- Back to 80x80 container size
                })
                
                local bgBounce = TweenService:Create(background, TweenInfo.new(
                    0.4, 
                    Enum.EasingStyle.Elastic, 
                    Enum.EasingDirection.Out
                ), {
                    Size = UDim2.fromOffset(72, 72)  -- Back to 80x80 container size
                })
                
                local strokeBounce = TweenService:Create(strokeContainer, TweenInfo.new(
                    0.4, 
                    Enum.EasingStyle.Elastic, 
                    Enum.EasingDirection.Out
                ), {
                    Size = UDim2.fromOffset(67, 67)  -- Back to 80x80 container size
                })
                
                local innerBounce = TweenService:Create(innerCircle, TweenInfo.new(
                    0.4, 
                    Enum.EasingStyle.Elastic, 
                    Enum.EasingDirection.Out
                ), {
                    Size = UDim2.fromOffset(60, 60)  -- Back to 80x80 container size
                })
                
                bounceTween:Play()
                bgBounce:Play()
                strokeBounce:Play()
                innerBounce:Play()
                
                -- Trigger the V keybind to open UI
                if UserInputService.KeyboardEnabled then
                    local VirtualInputManager = game:GetService("VirtualInputManager")
                    VirtualInputManager:SendKeyEvent(true, Enum.KeyCode.V, false, game)
                    task.wait(0.05)
                    VirtualInputManager:SendKeyEvent(false, Enum.KeyCode.V, false, game)
                end
            end
        end)
        -- ==================== END FIXED DRAGGING ====================
        
        -- Hover effects (adjusted for 80x80 container)
        button.MouseEnter:Connect(function()
            if not dragging then
                TweenService:Create(button, TweenInfo.new(0.2), {
                    Size = UDim2.fromOffset(66, 66)  -- Adjusted hover size for 80x80
                }):Play()
                TweenService:Create(background, TweenInfo.new(0.2), {
                    Size = UDim2.fromOffset(78, 78)  -- Adjusted hover size for 80x80
                }):Play()
                TweenService:Create(strokeContainer, TweenInfo.new(0.2), {
                    Size = UDim2.fromOffset(73, 73)  -- Adjusted hover size for 80x80
                }):Play()
                TweenService:Create(innerCircle, TweenInfo.new(0.2), {
                    Size = UDim2.fromOffset(66, 66)  -- Adjusted hover size for 80x80
                }):Play()
            end
        end)
        
        button.MouseLeave:Connect(function()
            if not dragging then
                TweenService:Create(button, TweenInfo.new(0.2), {
                    Size = UDim2.fromOffset(60, 60)  -- Back to normal size for 80x80
                }):Play()
                TweenService:Create(background, TweenInfo.new(0.2), {
                    Size = UDim2.fromOffset(72, 72)  -- Back to normal size for 80x80
                }):Play()
                TweenService:Create(strokeContainer, TweenInfo.new(0.2), {
                    Size = UDim2.fromOffset(67, 67)  -- Back to normal size for 80x80
                }):Play()
                TweenService:Create(innerCircle, TweenInfo.new(0.2), {
                    Size = UDim2.fromOffset(60, 60)  -- Back to normal size for 80x80
                }):Play()
            end
        end)
        
        toggleButton = screenGui
        toggleButtonEnabled = true
        
        -- Fade in animation
        button.ImageTransparency = 1
        background.BackgroundTransparency = 1
        strokeContainer.BackgroundTransparency = 1
        shadow.ImageTransparency = 1
        
        TweenService:Create(button, TweenInfo.new(0.3), {ImageTransparency = 0}):Play()
        TweenService:Create(background, TweenInfo.new(0.3), {BackgroundTransparency = 0.3}):Play()
        TweenService:Create(strokeContainer, TweenInfo.new(0.3), {BackgroundTransparency = 0}):Play()
        TweenService:Create(shadow, TweenInfo.new(0.3), {ImageTransparency = 0.6}):Play()
    end
end

local function removeToggleButton()
    if toggleButton then
        local container = toggleButton:FindFirstChild("Container")
        if container then
            local button = container:FindFirstChild("ToggleButton")
            local background = container:FindFirstChild("Background")
            if button then
                -- Fade out animation
                TweenService:Create(button, TweenInfo.new(0.2), {ImageTransparency = 1}):Play()
                local stroke = button:FindFirstChild("UIStroke")
                if stroke then
                    TweenService:Create(stroke, TweenInfo.new(0.2), {Transparency = 1}):Play()
                end
            end
            if background then
                TweenService:Create(background, TweenInfo.new(0.2), {BackgroundTransparency = 1}):Play()
            end
            task.wait(0.2)
        end
        toggleButton:Destroy()
        toggleButton = nil
        toggleButtonEnabled = false
    end
end

-- Monitor UI visibility state
task.spawn(function()
    local lastVisible = true
    while task.wait(0.5) do
        pcall(function()
            local uiVisible = false
            -- Check if the main UI container is visible
            for _, gui in pairs(game:GetService("CoreGui"):GetChildren()) do
                if gui.Name:find("Wind") or gui.Name:find("Da7mu") then
                    for _, child in pairs(gui:GetDescendants()) do
                        if child:IsA("Frame") and child.Visible and child.Size.X.Offset > 500 then
                            uiVisible = true
                            break
                        end
                    end
                end
            end
            
            -- Show toggle button only when UI is closed
            if not uiVisible and lastVisible then
                createToggleButton()
            elseif uiVisible and not lastVisible then
                removeToggleButton()
            end
            
            lastVisible = uiVisible
        end)
    end
end)
